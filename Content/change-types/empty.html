empty

The empty exists for testing purposes and adds an empty row to the DBCL table.

Examples

  • SQL example
  • YAML example
  • JSON example
  • XML example 
--liquibase formatted sql

--changeset liquibase-docs:empty-example
databaseChangeLog:
-  changeSet:
    id:  empty-example
    author:  liquibase-docs
    changes:
    -  empty:  {}
{

        {
            "changeSet": {
                "id": "empty-example",
                "author": "liquibase-docs",
                "changes": [
                    {
                        "empty": {}
                    }
                ]
            }
        }
    ]
}


    <changeSet  author="liquibase-docs"  id="empty-example">
        <empty/>
    </changeSet>

</databaseChangeLog>
