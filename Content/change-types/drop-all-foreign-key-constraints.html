dropAllForeignKeyConstraints

Drops all foreign key constraints for a table.

Available s

Name Description Required for Supports Since
baseTableCatalogName of the table containing columns constrained by foreign keys all 3.0
baseTableName containing columns constrained by foreign keys all all
baseTableSchemaName of the table containing columns constrained by foreign keys all
XML example 
<changeSet  author="liquibase-docs"  id="dropAllForeignKeyConstraints-example">  
    <dropAllForeignKeyConstraints  baseTableCatalogName="cat"  
            baseTableName="person"  
            baseTableSchemaName="public"/>  
</changeSet>
 YAML example 
-  changeSet:  
    id:  dropAllForeignKeyConstraints-example  
    author:  liquibase-docs  
    changes:  
    -  dropAllForeignKeyConstraints:  
        baseTableCatalogName:  cat  
        baseTableName:  person  
        baseTableSchemaName:  public
 JSON example 
{
  "changeSet":  {
    "id":  "dropAllForeignKeyConstraints-example",
    "author":  "liquibase-docs",
    "changes":  [{
      "dropAllForeignKeyConstraints":  {
        "baseTableCatalogName":  "cat",
        "baseTableName":  "person",
        "baseTableSchemaName":  "public"
      }
    }]
  }
}
