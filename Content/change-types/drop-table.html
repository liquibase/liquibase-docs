dropTable

Drops an existing table from your database.

Available s

Name Description Required for Supports Since
cascadeConstraints all
catalogName all 3.0
schemaName all
tableName

to drop

 all all
XML example 
<changeSet  author="liquibase-docs"  id="dropTable-example">  
    <dropTable  cascadeConstraints="true"  
            catalogName="cat"  
            schemaName="public"  
            tableName="person"/>  
</changeSet>
 YAML example 
-  changeSet:  
    id:  dropTable-example  
    author:  liquibase-docs  
    changes:  
    -  dropTable:  
        cascadeConstraints:  true  
        catalogName:  cat  
        schemaName:  public  
        tableName:  person
 JSON example 
{
  "changeSet":  {
    "id":  "dropTable-example",
    "author":  "liquibase-docs",
    "changes":  [{
      "dropTable":  {
        "cascadeConstraints":  true,
        "catalogName":  "cat",
        "schemaName":  "public",
        "tableName":  "person"
      }
    }]
  }
}
 SQL example 
DROP  TABLE  cat.person;
