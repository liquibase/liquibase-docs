renameSequence

Renames an existing sequence.

Available s

Name Description Required for Supports Since
catalogName all 3.0
newSequenceName New name for the sequence all all
oldSequenceName Existing name of the sequence to rename all all
schemaName all
XML example 
<changeSet  author="liquibase-docs"  id="renameSequence-example">  
    <renameSequence  catalogName="cat"  
            newSequenceName="seq_id"  
            oldSequenceName="seq_id"  
            schemaName="public"/>  
</changeSet>
 YAML example 
-  changeSet:  
    id:  renameSequence-example  
    author:  liquibase-docs  
    changes:  
    -  renameSequence:  
        catalogName:  cat  
        newSequenceName:  seq_id  
        oldSequenceName:  seq_id  
        schemaName:  public
 JSON example 
{
  "changeSet":  {
    "id":  "renameSequence-example",
    "author":  "liquibase-docs",
    "changes":  [{
      "renameSequence":  {
        "catalogName":  "cat",
        "newSequenceName":  "seq_id",
        "oldSequenceName":  "seq_id",
        "schemaName":  "public"
      }
    }]
  }
}
 SQL example 
sp_rename  seq_id  ,  
seq_id;
